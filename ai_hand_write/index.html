  <h1>Teachable Machine Model</h1>

  <button onclick="init()">Load Model</button>

  <p id="status">Model not loaded</p>

  <div id="output"></div>


  <!-- TensorFlow + Teachable Machine library -->

  <script src="https://cdn.jsdelivr.net/npm/@tensorflow/tfjs@latest"></script>

  <script src="https://cdn.jsdelivr.net/npm/@teachablemachine/image@latest"></script>


  <script>

    const URL = "./model/"; // folder where model files are stored

    let model, maxPredictions;


    async function init() {

      const modelURL = URL + "model.json";

      const metadataURL = URL + "metadata.json";


      model = await tmImage.load(modelURL, metadataURL);

      maxPredictions = model.getTotalClasses();


      document.getElementById("status").innerText = "Model loaded!";

      predict();

    }


    async function predict() {

      // For demo purposes we use a dummy image (you can replace with webcam or upload)

      const img = new Image();

      img.src = "https://teachablemachine.withgoogle.com/assets/img/default-image.png";


      img.onload = async () => {

        const prediction = await model.predict(img);


        let result = "";

        for (let i = 0; i < maxPredictions; i++) {

          result += prediction[i].className + ": " +

                    prediction[i].probability.toFixed(2) + "<br>";

        }


        document.getElementById("output").innerHTML = result;

      };

    }

  </script>
